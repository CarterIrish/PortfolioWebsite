127      /**
128       *
129       * @param n The selected number to factor
130       * @return
131       */
132      public String factorial(double n)
133      {
134          int fact=1;
135          for(int i = 1;i<=n;i++) // calculate the factorial of n or n!
136          {
137              fact=fact*i;
138          }
139          if (fact>99999999 || fact<=0) // if the number is too large ie. longer than 8 digits it will be inaccurate. so its replaced with *'s
140          {
141              return "*********";
142          }
143          return String.valueOf(fact);
144      }

146      /**
147       * Used to create the table with min & max values set using _minValue & _maxValue
148       */
149      public void start()
150      {
151          _rangeDisplay.setText("The range is : " +_tableMin+" to "+_tableMax);
152          Log.i("Carter", "Header row created");
153          // Row of headers
154          TableRow headerRow = new TableRow(this);
155          _parentTable.addView(headerRow);
156          // First column header
157          TextView header1 = new TextView(this);
158          header1.setText("Base Number");
159          headerRow.addView(header1);
160          // second column header
161          TextView header2 = new TextView(this);
162          header2.setText("Squared");
163          headerRow.addView(header2);
164          // third column header
165          TextView header3 = new TextView(this);
166          header3.setText("Cubed");
167          headerRow.addView(header3);
168          // fourth column header
169          TextView header4 = new TextView(this);
170          header4.setText("Sqrt");
171          headerRow.addView(header4);
172          // fifth column
173          TextView header5 = new TextView(this);
174          header5.setText("Fact");
175          headerRow.addView(header5);
176          // The data cells
177          for(double i=this._tableMin;i<(this._tableMax+1);i++)
178          {
179              // adding in the row
180              Log.i("Carter", "Row Created");
181                  TableRow row = new TableRow(this);
182                  _parentTable.addView(row);
183              // filling in the first column
184              Log.i("Carter", "new TextView added #1");
185                  TextView textView1 = new TextView(this);
186                  textView1.setText(""+i);
187                  row.addView(textView1);
188              // filling in the second column
189              Log.i("Carter", "new TextView added #2");
190                  TextView textView2 = new TextView(this);
191                  textView2.setText(String.format("%.2f", Math.pow(i,2)));
192                  row.addView(textView2);
193              // filling in the third column
194              Log.i("Carter", "new TextView added #3");
195                  TextView textView3 = new TextView(this);
196                  textView3.setText(String.format("%.2f", Math.pow(i,3)));
197                  row.addView(textView3);
198              // filling in the fourth column
199              Log.i("Carter", "new TextView added #4");
200                  TextView textView4 = new TextView(this);
201                  textView4.setText(String.format("%.2f", Math.sqrt(i)));
202                  row.addView(textView4);
203              // filling in the fifth column
204              Log.i("Carter", "new TextView added #5");
205                  TextView textView5 = new TextView(this);
206                  textView5.setText(factorial(i));
207                  row.addView(textView5);
208          }

211      /**
212       *
213       * @param sharedPreferences The {@link SharedPreferences} that received the change.
214       * @param key The key of the preference that was changed, added, or removed. Apps targeting
215       *            {@link android.os.Build.VERSION_CODES#R} on devices running OS versions
216       *            {@link android.os.Build.VERSION_CODES#R Android R} or later, will receive
217       *            a {@code null} value when preferences are cleared.
218       */
219      @Override
220      public void onSharedPreferenceChanged(SharedPreferences sharedPreferences, String key)
221      {
222          Log.i("Carter", "key changed "+key);
223          switch(key)
224          {
225              // called if the baseNumCol preference is changed
226              case "baseNumCol":
227                  Boolean showNumber0 = _preferences.getBoolean(key,true);
228                  Log.i("carter","Bool val: "+_baseCol);
229                  break;
230              // called if the sqrCol preference is changed
231              case "sqrCol":
232                  Boolean showNumber1 = _preferences.getBoolean(key, true);
233                  Log.i("carter", "Bool val: "+_sqrCol);
234                  _parentTable.setColumnCollapsed(1,!showNumber1);
235                  break;
236              // called if the sqrtCol preference is changed
237              case "sqrtCol":
238                  Boolean showNumber2 = _preferences.getBoolean(key,true);
239                  Log.i("carter", "Bool val: "+_sqrtCol);
240                  _parentTable.setColumnCollapsed(3,!showNumber2);
241                  break;
242              // called if the cubeCol preference is changed
243              case "cubeCol":
244                  Boolean showNumber3 = _preferences.getBoolean(key,true);
245                  Log.i("carter", "Bool val: "+_cubeCol);
246                  _parentTable.setColumnCollapsed(2,!showNumber3);
247                  break;
248              // called if the factorialCol preference is changed
249              case "factorialCol":
250                  Boolean showNumber4 = _preferences.getBoolean(key,true);
251                  Log.i("carter","Bool val: "+_factorialCol);
252                  _parentTable.setColumnCollapsed(4,!showNumber4);
253                  break;
254              // called if the maxValue preference is changed
255              case "maxValue":
256                  _tableMax = Integer.valueOf(_preferences.getString("maxValue","10"));
257                  start(); // call start() because I changed the range which means I need to redo the table instead of just hiding the columns
258                  break;
259              // called if the minValue preference is changed
260              case "minValue":
261                  _tableMin = Integer.valueOf(_preferences.getString("minValue","1"));
262                  start(); // call start() because I changed the range which means I need to redo the table instead of just hiding the columns
263                  break;
264          }
265      }
